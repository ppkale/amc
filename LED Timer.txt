#include void  
delay(); 
int main(void)  
{ 
IO0DIR=0xffffffff; 
VPBDIV=0x01; //Fosc=12Mhz,CCLK=60Mhz,PCLK=60MHz  
while(1)  
{ 
IO0SET=0xffffffff;  
delay(); //1sec delay  
IO0CLR=0xffffffff; 
delay(); //1sec delay 
} 
} 
void delay()  
{ 
T1CTCR=0x0; //Select Timer Mode  
T1TCR=0x00; //Timer OFF 
T1MR0=1000; //Match Register Value 
T1MCR=0x00000004;  
//TC and PC will be stopped and TCR[0] will be set to 0 if MR0 matches the TC.  T1PR=59999; //Prescaler value for 1ms 
T1TCR=0x02; //Timer reset  
T1TCR=0x01; //Timer ON  
while(T1TC!=T1MR0); //1sec delay 
T1TCR=0x00; //Timer OFF
